68 template <typename FloatType>
69 class uniform_floating_point_distribution {
70  static_assert(std::is_floating_point<FloatType>::value, "...");
71  static_assert(!std::is_same<FloatType, long double>::value,
72  "We do not support long double due to inconsistent behaviour between platforms");
73 
74  using WidthType = Detail::DistanceType<FloatType>;
75 
76  FloatType m_a, m_b;
77  FloatType m_ulp_magnitude;
78  WidthType m_floats_in_range;
79  uniform_integer_distribution<WidthType> m_int_dist;
80 
81  // In specific cases, we can overflow into `inf` when computing the
82  // `steps * g` offset. To avoid this, we don't offset by more than this
83  // in one multiply + addition.
84  WidthType m_max_steps_in_one_go;
85  // We don't want to do the magnitude check every call to `operator()`
86  bool m_a_has_leq_magnitude;
87 
88 public:
89  using result_type = FloatType;
90 
91  uniform_floating_point_distribution( FloatType a, FloatType b ):
92  m_a( a ),
93  m_b( b ),
94  m_ulp_magnitude( Detail::gamma( m_a, m_b ) ),
95  m_floats_in_range( Detail::count_equidistant_floats( m_a, m_b, m_ulp_magnitude ) ),
96  m_int_dist(0, m_floats_in_range),
97  m_max_steps_in_one_go( Detail::calculate_max_steps_in_one_go(m_ulp_magnitude)),
98  m_a_has_leq_magnitude(std::fabs(m_a) <= std::fabs(m_b))
99  {
100  assert( a <= b );
101  }
102 
103  template <typename Generator>
104  result_type operator()( Generator& g ) {
105  WidthType steps = m_int_dist( g );
106  if ( m_a_has_leq_magnitude ) {
107  if ( steps == m_floats_in_range ) { return m_a; }
108  auto b = m_b;
109  while (steps > m_max_steps_in_one_go) {
110  b -= m_max_steps_in_one_go * m_ulp_magnitude;
111  steps -= m_max_steps_in_one_go;
112  }
113  return b - steps * m_ulp_magnitude;
114  } else {
115  if ( steps == m_floats_in_range ) { return m_b; }
116  auto a = m_a;
117  while (steps > m_max_steps_in_one_go) {
118  a += m_max_steps_in_one_go * m_ulp_magnitude;
119  steps -= m_max_steps_in_one_go;
120  }
121  return a + steps * m_ulp_magnitude;
122  }
123  }
124 
125  result_type a() const { return m_a; }
126  result_type b() const { return m_b; }
127 };
