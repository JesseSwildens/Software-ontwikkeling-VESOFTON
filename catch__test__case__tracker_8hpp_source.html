21  struct NameAndLocation {
22  std::string name;
23  SourceLineInfo location;
24 
25  NameAndLocation( std::string&& _name, SourceLineInfo const& _location );
26  friend bool operator==(NameAndLocation const& lhs, NameAndLocation const& rhs) {
27  // This is a very cheap check that should have a very high hit rate.
28  // If we get to SourceLineInfo::operator==, we will redo it, but the
29  // cost of repeating is trivial at that point (we will be paying
30  // multiple strcmp/memcmps at that point).
31  if ( lhs.location.line != rhs.location.line ) { return false; }
32  return lhs.name == rhs.name && lhs.location == rhs.location;
33  }
34  friend bool operator!=(NameAndLocation const& lhs,
35  NameAndLocation const& rhs) {
36  return !( lhs == rhs );
37  }
38  };
