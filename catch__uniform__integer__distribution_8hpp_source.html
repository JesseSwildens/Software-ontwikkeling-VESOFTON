27 template <typename IntegerType>
28 class uniform_integer_distribution {
29  static_assert(std::is_integral<IntegerType>::value, "...");
30 
31  using UnsignedIntegerType = Detail::SizedUnsignedType_t<sizeof(IntegerType)>;
32 
33  // Only the left bound is stored, and we store it converted to its
34  // unsigned image. This avoids having to do the conversions inside
35  // the operator(), at the cost of having to do the conversion in
36  // the a() getter. The right bound is only needed in the b() getter,
37  // so we recompute it there from other stored data.
38  UnsignedIntegerType m_a;
39 
40  // How many different values are there in [a, b]. a == b => 1, can be 0 for distribution over all values in the type.
41  UnsignedIntegerType m_ab_distance;
42 
43  // We hoisted this out of the main generation function. Technically,
44  // this means that using this distribution will be slower than Lemire's
45  // algorithm if this distribution instance will be used only few times,
46  // but it will be faster if it is used many times. Since Catch2 uses
47  // distributions only to implement random generators, we assume that each
48  // distribution will be reused many times and this is an optimization.
49  UnsignedIntegerType m_rejection_threshold = 0;
50 
51  UnsignedIntegerType computeDistance(IntegerType a, IntegerType b) const {
52  // This overflows and returns 0 if a == 0 and b == TYPE_MAX.
53  // We handle that later when generating the number.
54  return transposeTo(b) - transposeTo(a) + 1;
55  }
56 
57  static UnsignedIntegerType computeRejectionThreshold(UnsignedIntegerType ab_distance) {
58  // distance == 0 means that we will return all possible values from
59  // the type's range, and that we shouldn't reject anything.
60  if ( ab_distance == 0 ) { return 0; }
61  return ( ~ab_distance + 1 ) % ab_distance;
62  }
63 
64  static UnsignedIntegerType transposeTo(IntegerType in) {
65  return Detail::transposeToNaturalOrder<IntegerType>(
66  static_cast<UnsignedIntegerType>( in ) );
67  }
68  static IntegerType transposeBack(UnsignedIntegerType in) {
69  return static_cast<IntegerType>(
70  Detail::transposeToNaturalOrder<IntegerType>(in) );
71  }
72 
73 public:
74  using result_type = IntegerType;
75 
76  uniform_integer_distribution( IntegerType a, IntegerType b ):
77  m_a( transposeTo(a) ),
78  m_ab_distance( computeDistance(a, b) ),
79  m_rejection_threshold( computeRejectionThreshold(m_ab_distance) ) {
80  assert( a <= b );
81  }
82 
83  template <typename Generator>
84  result_type operator()( Generator& g ) {
85  // All possible values of result_type are valid.
86  if ( m_ab_distance == 0 ) {
87  return transposeBack( Detail::fillBitsFrom<UnsignedIntegerType>( g ) );
88  }
89 
90  auto random_number = Detail::fillBitsFrom<UnsignedIntegerType>( g );
91  auto emul = Detail::extendedMult( random_number, m_ab_distance );
92  // Unlike Lemire's algorithm we skip the ab_distance check, since
93  // we precomputed the rejection threshold, which is always tighter.
94  while (emul.lower < m_rejection_threshold) {
95  random_number = Detail::fillBitsFrom<UnsignedIntegerType>( g );
96  emul = Detail::extendedMult( random_number, m_ab_distance );
97  }
98 
99  return transposeBack(m_a + emul.upper);
100  }
101 
102  result_type a() const { return transposeBack(m_a); }
103  result_type b() const { return transposeBack(m_ab_distance + m_a - 1); }
104 };
