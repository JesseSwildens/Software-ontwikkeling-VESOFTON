83  constexpr ExtendedMultResult<std::uint64_t>
84  extendedMultPortable(std::uint64_t lhs, std::uint64_t rhs) {
85 #define CarryBits( x ) ( x >> 32 )
86 #define Digits( x ) ( x & 0xFF'FF'FF'FF )
87  std::uint64_t lhs_low = Digits( lhs );
88  std::uint64_t rhs_low = Digits( rhs );
89  std::uint64_t low_low = ( lhs_low * rhs_low );
90  std::uint64_t high_high = CarryBits( lhs ) * CarryBits( rhs );
91 
92  // We add in carry bits from low-low already
93  std::uint64_t high_low =
94  ( CarryBits( lhs ) * rhs_low ) + CarryBits( low_low );
95  // Note that we can add only low bits from high_low, to avoid
96  // overflow with large inputs
97  std::uint64_t low_high =
98  ( lhs_low * CarryBits( rhs ) ) + Digits( high_low );
99 
100  return { high_high + CarryBits( high_low ) + CarryBits( low_high ),
101  ( low_high << 32 ) | Digits( low_low ) };
102 #undef CarryBits
103 #undef Digits
104  }

139  template <typename TargetType,
140  typename Generator>
141  std::enable_if_t<sizeof(typename Generator::result_type) >= sizeof(TargetType),
142  TargetType> fillBitsFrom(Generator& gen) {
143  using gresult_type = typename Generator::result_type;
144  static_assert( std::is_unsigned<TargetType>::value, "Only unsigned integers are supported" );
145  static_assert( Generator::min() == 0 &&
146  Generator::max() == static_cast<gresult_type>( -1 ),
147  "Generator must be able to output all numbers in its result type (effectively it must be a random bit generator)" );
148 
149  // We want to return the top bits from a generator, as they are
150  // usually considered higher quality.
151  constexpr auto generated_bits = sizeof( gresult_type ) * CHAR_BIT;
152  constexpr auto return_bits = sizeof( TargetType ) * CHAR_BIT;
153 
154  return static_cast<TargetType>( gen() >>
155  ( generated_bits - return_bits) );
156  }

158  template <typename TargetType,
159  typename Generator>
160  std::enable_if_t<sizeof(typename Generator::result_type) < sizeof(TargetType),
161  TargetType> fillBitsFrom(Generator& gen) {
162  using gresult_type = typename Generator::result_type;
163  static_assert( std::is_unsigned<TargetType>::value,
164  "Only unsigned integers are supported" );
165  static_assert( Generator::min() == 0 &&
166  Generator::max() == static_cast<gresult_type>( -1 ),
167  "Generator must be able to output all numbers in its result type (effectively it must be a random bit generator)" );
168 
169  constexpr auto generated_bits = sizeof( gresult_type ) * CHAR_BIT;
170  constexpr auto return_bits = sizeof( TargetType ) * CHAR_BIT;
171  std::size_t filled_bits = 0;
172  TargetType ret = 0;
173  do {
174  ret <<= generated_bits;
175  ret |= gen();
176  filled_bits += generated_bits;
177  } while ( filled_bits < return_bits );
178 
179  return ret;
180  }

182  /*
183  * Transposes numbers into unsigned type while keeping their ordering
184  *
185  * This means that signed types are changed so that the ordering is
186  * [INT_MIN, ..., -1, 0, ..., INT_MAX], rather than order we would
187  * get by simple casting ([0, ..., INT_MAX, INT_MIN, ..., -1])
188  */
189  template <typename OriginalType, typename UnsignedType>
190  std::enable_if_t<std::is_signed<OriginalType>::value, UnsignedType>
191  transposeToNaturalOrder( UnsignedType in ) {
192  static_assert(
193  sizeof( OriginalType ) == sizeof( UnsignedType ),
194  "reordering requires the same sized types on both sides" );
195  static_assert( std::is_unsigned<UnsignedType>::value,
196  "Input type must be unsigned" );
197  // Assuming 2s complement (standardized in current C++), the
198  // positive and negative numbers are already internally ordered,
199  // and their difference is in the top bit. Swapping it orders
200  // them the desired way.
201  constexpr auto highest_bit =
202  UnsignedType( 1 ) << ( sizeof( UnsignedType ) * CHAR_BIT - 1 );
203  return static_cast<UnsignedType>( in ^ highest_bit );
204  }
