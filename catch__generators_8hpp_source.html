206  template<typename L>
207  auto generate( StringRef generatorName, SourceLineInfo const& lineInfo, L const& generatorExpression ) -> typename decltype(generatorExpression())::type {
208  using UnderlyingType = typename decltype(generatorExpression())::type;
209 
210  IGeneratorTracker* tracker = acquireGeneratorTracker( generatorName, lineInfo );
211  // Creation of tracker is delayed after generator creation, so
212  // that constructing generator can fail without breaking everything.
213  if (!tracker) {
214  tracker = createGeneratorTracker(
215  generatorName,
216  lineInfo,
217  Catch::Detail::make_unique<Generators<UnderlyingType>>(
218  generatorExpression() ) );
219  }
220 
221  auto const& generator = static_cast<IGenerator<UnderlyingType> const&>( *tracker->getGenerator() );
222  return generator.get();
223  }
