100  template <typename ForwardIter1,
101  typename Sentinel1,
102  typename ForwardIter2,
103  typename Sentinel2,
104  typename Comparator>
105  bool is_permutation( ForwardIter1 first_1,
106  const Sentinel1 end_1,
107  ForwardIter2 first_2,
108  const Sentinel2 end_2,
109  Comparator cmp ) {
110  // TODO: no optimization for stronger iterators, because we would also have to constrain on sentinel vs not sentinel types
111  // TODO: Comparator has to be "both sides", e.g. a == b => b == a
112  // This skips shared prefix of the two ranges
113  while (first_1 != end_1 && first_2 != end_2 && cmp(*first_1, *first_2)) {
114  ++first_1;
115  ++first_2;
116  }
117 
118  // We need to handle case where at least one of the ranges has no more elements
119  if (first_1 == end_1 || first_2 == end_2) {
120  return first_1 == end_1 && first_2 == end_2;
121  }
122 
123  // pair counting is n**2, so we pay linear walk to compare the sizes first
124  auto dist_1 = sentinel_distance( first_1, end_1 );
125  auto dist_2 = sentinel_distance( first_2, end_2 );
126 
127  if (dist_1 != dist_2) { return false; }
128 
129  // Since we do not try to handle stronger iterators pair (e.g.
130  // bidir) optimally, the only thing left to do is to check counts in
131  // the remaining ranges.
132  return check_element_counts( first_1, end_1, first_2, end_2, cmp );
133  }
