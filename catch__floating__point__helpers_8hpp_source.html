59  template <typename FP>
60  uint64_t ulpDistance( FP lhs, FP rhs ) {
61  assert( std::numeric_limits<FP>::is_iec559 &&
62  "ulpDistance assumes IEEE-754 format for floating point types" );
63  assert( !Catch::isnan( lhs ) &&
64  "Distance between NaN and number is not meaningful" );
65  assert( !Catch::isnan( rhs ) &&
66  "Distance between NaN and number is not meaningful" );
67 
68  // We want X == Y to imply 0 ULP distance even if X and Y aren't
69  // bit-equal (-0 and 0), or X - Y != 0 (same sign infinities).
70  if ( lhs == rhs ) { return 0; }
71 
72  // We need a properly typed positive zero for type inference.
73  static constexpr FP positive_zero{};
74 
75  // We want to ensure that +/- 0 is always represented as positive zero
76  if ( lhs == positive_zero ) { lhs = positive_zero; }
77  if ( rhs == positive_zero ) { rhs = positive_zero; }
78 
79  // If arguments have different signs, we can handle them by summing
80  // how far are they from 0 each.
81  if ( std::signbit( lhs ) != std::signbit( rhs ) ) {
82  return ulpDistance( std::abs( lhs ), positive_zero ) +
83  ulpDistance( std::abs( rhs ), positive_zero );
84  }
85 
86  // When both lhs and rhs are of the same sign, we can just
87  // read the numbers bitwise as integers, and then subtract them
88  // (assuming IEEE).
89  uint64_t lc = Detail::convertToBits( lhs );
90  uint64_t rc = Detail::convertToBits( rhs );
91 
92  // The ulp distance between two numbers is symmetric, so to avoid
93  // dealing with overflows we want the bigger converted number on the lhs
94  if ( lc < rc ) {
95  std::swap( lc, rc );
96  }
97 
98  return lc - rc;
99  }
