44 #define INTERNAL_CATCH_TEST( macroName, resultDisposition, ... ) \
45  do { /* NOLINT(bugprone-infinite-loop) */ \
46  /* The expression should not be evaluated, but warnings should hopefully be checked */ \
47  CATCH_INTERNAL_IGNORE_BUT_WARN(__VA_ARGS__); \
48  Catch::AssertionHandler catchAssertionHandler( macroName##_catch_sr, CATCH_INTERNAL_LINEINFO, CATCH_INTERNAL_STRINGIFY(__VA_ARGS__), resultDisposition ); \
49  INTERNAL_CATCH_TRY { \
50  CATCH_INTERNAL_START_WARNINGS_SUPPRESSION \
51  CATCH_INTERNAL_SUPPRESS_PARENTHESES_WARNINGS \
52  catchAssertionHandler.handleExpr( Catch::Decomposer() <= __VA_ARGS__ ); /* NOLINT(bugprone-chained-comparison) */ \
53  CATCH_INTERNAL_STOP_WARNINGS_SUPPRESSION \
54  } INTERNAL_CATCH_CATCH( catchAssertionHandler ) \
55  INTERNAL_CATCH_REACT( catchAssertionHandler ) \
56  } while( (void)0, (false) && static_cast<const bool&>( !!(__VA_ARGS__) ) ) // the expression here is never evaluated at runtime but it forces the compiler to give it a look
57  // The double negation silences MSVC's C4800 warning, the static_cast forces short-circuit evaluation if the type has overloaded &&.
